function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(
      '1Umj26c76Gx-0fkxgH-eD1v0HE1ItBHOx7HVbC1AGpBA');
    var sh = ss.getSheetByName('Phlebitis');
    if (!sh) {
      sh = ss.insertSheet('Phlebitis');
      var hdr = ['ID','Timestamp','HN','ชื่อ-สกุล',
        'Ward','Bed','ตำแหน่ง IV','VIP Score',
        'Grade','อาการ','ระยะเวลา(ชม.)','สารน้ำ/ยา',
        'พยาบาล','หมายเหตุ','AI วิเคราะห์'];
      sh.appendRow(hdr);
      sh.getRange(1,1,1,hdr.length)
        .setFontWeight('bold')
        .setBackground('#2563EB')
        .setFontColor('#fff');
      sh.setFrozenRows(1);
    }
    if (d._action === 'update' && d._rowIndex) {
      var row = parseInt(d._rowIndex) + 1;
      sh.getRange(row,2,1,14).setValues([[
        d.ts,d.hn,d.name,d.ward,d.bed,
        d.site,d.vip,d.grade,d.syms,
        d.hours,d.drugs,d.nurse,
        d.notes,d.aiText||'']]);
      return ContentService.createTextOutput(
        JSON.stringify({ok:true,action:'updated'}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    if (d._action === 'delete' && d._rowIndex) {
      sh.deleteRow(parseInt(d._rowIndex)+1);
      return ContentService.createTextOutput(
        JSON.stringify({ok:true,action:'deleted'}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    var newId = 'PH'+new Date().getTime();
    sh.appendRow([newId,d.ts,d.hn,d.name,
      d.ward,d.bed,d.site,d.vip,d.grade,
      d.syms,d.hours,d.drugs,d.nurse,
      d.notes,d.aiText||'']);
    return ContentService.createTextOutput(
      JSON.stringify({ok:true,id:newId,
        row:sh.getLastRow()}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({ok:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
  try {
    var ss = SpreadsheetApp.openById(
      '1Umj26c76Gx-0fkxgH-eD1v0HE1ItBHOx7HVbC1AGpBA');
    var sh = ss.getSheetByName('Phlebitis');
    if (!sh) return ContentService.createTextOutput(
      JSON.stringify({ok:true,rows:[]}))
      .setMimeType(ContentService.MimeType.JSON);
    var vals = sh.getDataRange().getValues();
    return ContentService.createTextOutput(
      JSON.stringify({ok:true,rows:vals}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({ok:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
