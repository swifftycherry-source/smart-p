function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(
      '1Umj26c76Gx-0fkxgH-eD1v0HE1ItBHOx7HVbC1AGpBA');
    var sh = ss.getSheetByName('SMART-P');
    if (!sh) {
      sh = ss.insertSheet('SMART-P');
      var h=['ID','Timestamp','ประเภท','HN',
        'ชื่อ-สกุล','Ward','Bed','ตำแหน่ง IV',
        'Grade','ระดับ','อาการ','ระยะเวลา(ชม.)',
        'สารน้ำ/ยา','พยาบาล','หมายเหตุ',
        'AI วิเคราะห์','แผนพยาบาล','รูปภาพ'];
      sh.appendRow(h);
      sh.getRange(1,1,1,h.length)
        .setFontWeight('bold')
        .setBackground('#1D4ED8')
        .setFontColor('#fff');
      sh.setFrozenRows(1);
    }
    if (d._action==='update' && d._ri) {
      sh.getRange(parseInt(d._ri)+1,2,1,17)
        .setValues([[d.ts,d.type,d.hn,d.name,
          d.ward,d.bed,d.site,d.grade,d.gradeLbl,
          d.syms,d.hours,d.drugs,d.nurse,d.notes,
          d.aiTxt,d.npTxt,d.img||'']]);
      return ok({action:'updated'});
    }
    if (d._action==='delete'&&d._ri) {
      sh.deleteRow(parseInt(d._ri)+1);
      return ok({action:'deleted'});
    }
    var id='SMARTP-'+Date.now();
    sh.appendRow([id,d.ts,d.type,d.hn,d.name,
      d.ward,d.bed,d.site,d.grade,d.gradeLbl,
      d.syms,d.hours,d.drugs,d.nurse,d.notes,
      d.aiTxt,d.npTxt,d.img||'']);
    // Insert image in col 18 if exists
    if (d.img && d.img.length > 100) {
      try {
        var imgData = Utilities.base64Decode(d.img);
        var blob = Utilities.newBlob(imgData,'image/jpeg','iv_site.jpg');
        var folder = DriveApp.getRootFolder();
        var file = folder.createFile(blob);
        file.setSharing(DriveApp.Access.ANYONE,DriveApp.Permission.VIEW);
        var imgUrl = 'https://drive.google.com/uc?id='+file.getId();
        sh.getRange(sh.getLastRow(),18).setValue('=IMAGE("'+imgUrl+'",4,60,80)');
      } catch(imgErr){}
    }
    return ok({id:id,row:sh.getLastRow()});
  } catch(err){return err_(err.toString());}
}
function doGet(e){
  try{
    var ss=SpreadsheetApp.openById(
      '1Umj26c76Gx-0fkxgH-eD1v0HE1ItBHOx7HVbC1AGpBA');
    var sh=ss.getSheetByName('SMART-P');
    if(!sh)return ok({rows:[]});
    return ok({rows:sh.getDataRange().getValues()});
  }catch(err){return err_(err.toString());}
}
function ok(d){return ContentService.createTextOutput(JSON.stringify(Object.assign({ok:true},d))).setMimeType(ContentService.MimeType.JSON);}
function err_(m){return ContentService.createTextOutput(JSON.stringify({ok:false,error:m})).setMimeType(ContentService.MimeType.JSON);}
